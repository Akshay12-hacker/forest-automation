            if (isSearchOpen) {
                $('#searchPark').hide();
                $('#SearchDiv').removeClass('MainSearchInitiatorImgEnlarge');
                $('#SearchDiv').removeClass('MainSearchInitiatorImgEnlargeMobile');
                $('#SearchDiv').addClass('MainSearchInitiatorImgNormal');

                $('#searchIcon').attr('src', imgSearch.attr('src'));
                isSearchOpen = false;
                $('#divResult').empty();
                $('#searchPark').val('');
            }
            else {
                $('#searchPark').show();
                $('#SearchDiv').removeClass('MainSearchInitiatorImgNormal');
                if (window.innerWidth < 767) {
                    $('#SearchDiv').addClass('MainSearchInitiatorImgEnlargeMobile');
                } else { $('#SearchDiv').addClass('MainSearchInitiatorImgEnlarge'); }

                $('#searchIcon').attr('src', imgClose.attr('src'));
                $('#searchPark').focus();
                isSearchOpen = true;

        /* General Search  */
        function showSearchResult() {
            //debugger;
            var Sval = $('#searchPark').val();
            if (Sval.length > 2) {

                $.ajax({
                    type: "POST",
                    url: "index.aspx/GeneralSearch",
                    data: '{SearchText: "' + Sval + '" }',
                    contentType: "application/json; charset=utf-8",
                    dataType: "json",
                    success: OnSuccess,
                    failure: function (response) {
                        alert(response.d);
                    }
                });
            } else {
                $('#divResult').empty();
            }
        }

        function OnSuccess(response) {
            /*debugger;*/
            var ulObj = $('<ul>');

            var result = JSON.parse(response.d);
            var resultObj = result["SearchResult"];
            for (var i = 0; i < resultObj.length; i++) {
                var li = $('<li>');
                var ParkName = resultObj[i]["ParkName"];
                var IsBookingAvailable = JSON.parse(resultObj[i]["IsBookingAvailable"].toLowerCase());
                var IsBrochureAvailable = JSON.parse(resultObj[i]["IsBrochureAvailable"].toLowerCase());
                var IsNightSafariBookingAvailable = JSON.parse(resultObj[i]["IsNightSafariBookingAvailable"].toLowerCase());;
                var BookingLink = resultObj[i]["BookingLink"];
                var NightSafariBookingLink = resultObj[i]["NightSafariBookingLink"];
                var BrochureLink = resultObj[i]["BrochureLink"];
                var IsWebSiteAvailable = JSON.parse(resultObj[i]["IsWebSiteAvailable"].toLowerCase());
                var WebSiteLink = resultObj[i]["WebSiteLink"];
                var span = $('<span>' + ParkName + '</span>');
                li.append(span);
                if (IsBookingAvailable) {
                    //var linkBrochure = $('<a>');
                    //linkBrochure.attr({ 'href': BrochureLink, 'target': '_blank' });
                    var BookingText = 'Book Now';
                    if (IsNightSafariBookingAvailable) { BookingText = 'Book Day Safari'; }
                    li.append($('<a>').attr({ 'href': BookingLink, 'target': '_self' }).text(BookingText).addClass('searchResultAnchorPrimary'));
                }


                if (IsBrochureAvailable) {
                    //var linkBrochure = $('<a>');
                    //linkBrochure.attr({ 'href': BrochureLink, 'target': '_blank' });
                    li.append($('<a>').attr({ 'href': BrochureLink, 'target': '_blank' }).text('Brochure').addClass('searchResultAnchorRed'));
                }

                if (IsNightSafariBookingAvailable) {
                    //var linkBrochure = $('<a>');
                    //linkBrochure.attr({ 'href': BrochureLink, 'target': '_blank' });
                    li.append($('<a>').attr({ 'href': NightSafariBookingLink, 'target': '_self' }).text('Book Night Safari').addClass('searchResultAnchorSecondary'));
                }

                if (IsWebSiteAvailable) {

                    li.append($('<a>').attr({ 'href': WebSiteLink, 'target': '_blank' }).text('WebSite').addClass('searchResultAnchorRed'));
                }

                li.addClass("searchResultLi");
                ulObj.append(li);
            }
            ulObj.addClass("searchResultUl");
            $('#divResult').empty();
            $('#divResult').html(ulObj);

        }

        document.addEventListener('keydown', function (event) {
            // F12
            if (event.key === "F12" || event.keyCode === 123) {
                event.preventDefault();
                return false;
            }

            // Ctrl+Shift+I, Ctrl+Shift+J, Ctrl+U
            if ((event.ctrlKey && event.shiftKey && (event.key === 'I' || event.key === 'J')) ||
                (event.ctrlKey && event.key === 'U')) {
                event.preventDefault();
                return false;
            }
        });

        // Detect DevTools open
        (function () {
            let devtoolsOpen = false;
            const threshold = 160;

            function detectDevTools() {
                const widthThreshold = window.outerWidth - window.innerWidth > threshold;
                const heightThreshold = window.outerHeight - window.innerHeight > threshold;
                return widthThreshold || heightThreshold;
            }

            function check() {
                if (detectDevTools()) {
                    if (!devtoolsOpen) {
                        devtoolsOpen = true;
                        alert('Developer tools are open!');
                        window.location.href = "https://forest.mponline.gov.in/Index.aspx"; // Take action
                    }
                } else {
                    devtoolsOpen = false;
                }
            }

            setInterval(check, 1000); // check every second
        })();

    </script>
